% -----------------------------------------------------------------------
% Local styles (extend preamble's rbox / connect)
% -----------------------------------------------------------------------
\tikzset{
  % Parameterised zone boundary — call as zonebox=teal or zonebox=gray
  zonebox/.style={
    rbox, rounded corners=10pt, inner sep=14pt, line width=0.6pt,
    draw=#1!50!black, fill=#1!8,
  },
  zonebox/.default=gray,
  % Parameterised service card — call as svcbox=teal or svcbox=gray
  svcbox/.style={
    rbox, rounded corners=5pt, inner sep=6pt, line width=0.4pt,
    text width=3.2cm, align=center, minimum height=1.1cm,
    draw=#1!50!black, fill=#1!8,
  },
  svcbox/.default=gray,
  % Pending (dashed) service card
  svcpending/.style={svcbox=orange, dashed},
  % Thin grey arrows
  xarr/.style={connect, line width=0.5pt},
  xarrpending/.style={xarr, dashed},
  % Text helpers
  zonelabel/.style={font=\footnotesize\bfseries},
  sublabel/.style={font=\scriptsize, text=gray},
}
